# Button

Button component with different props.

#### Example

<Canvas of={Button.Primary} />

## Usage

```ts
import {Button} from "bagheera-ui";

const Example = () => {
  return (
      <Button
        size="medium"
        text="Button"
        type="primary"
      />
  );
};

export default Example;
```

#### Arguments


<table>
  <tr>
    <th>Argument</th>
    <th>Description</th>
    <th>Type</th>
    <th>Required</th>
  </tr>
  <tr>
    <td>text</td>
    <td>The text content of the button</td>
    <td>string</td>
    <td>Yes</td>
  </tr>
  <tr>
    <td>type</td>
    <td>Indicates what kind of styling the button has</td>
    <td>'primary' | 'secondary' | 'tertiary' | 'lightning' | 'primary inverse' | 'secondary inverse' | 'tertiary inverse' | 'negative'</td>
    <td>Yes</td>
  </tr>
  <tr>
    <td>size</td>
    <td>Defines the size of the button</td>
    <td>'small' | 'medium' | 'large'</td>
    <td>Yes</td>
  </tr>
  <tr>
    <td>onClick</td>
    <td>A function that is called when the button is clicked. It receives a MouseEventHandler for handling the click event on the button element</td>
    <td>`(e?: MouseEvent) => void;`</td>
    <td>No</td>
  </tr>
  <tr>
    <td>width</td>
    <td>A custom css width for the button</td>
    <td>string</td>
    <td>No</td>
  </tr>
  <tr>
    <td>disabled</td>
    <td>Whether the button should be disabled or not</td>
    <td>boolean</td>
    <td>No</td>
  </tr>
  <tr>
    <td>loading</td>
    <td>If the button should display a loading state</td>
    <td>boolean</td>
    <td>No</td>
  </tr>
  <tr>
    <td>selected</td>
    <td>If the button should display a selected state</td>
    <td>boolean</td>
    <td>No</td>
  </tr>
  <tr>
    <td>icon</td>
    <td>Will prefix button content with an icon from the icon library</td>
    <td>`IconNames`</td>
    <td>No</td>
  </tr>
</table>
